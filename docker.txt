# Получить набор доступных команд в докере
docker --help

# Получить набор доступных опций для выбранной команды.
docker <COMMAND> --help

# Запустить контейнер с параметрами и определеным именем.
# Если такого конейнера нет, но он есть в библиотеке докера 
# то перед запуском его скачивает с репозитория докера, а затем запускает этот контейнер.
docker run -d -p 80:80 docker/getting-started 

# Связывание порта машины с портом контейнера
# Локальная машина : контейнер
-p 80:80 

# Полная информация по нашему Докеру на машине
docker info

# Вывести все контейнеры которые есть на машине
docker ps

# Остановить один или несколько запущенных контейнеров
docker stop <CONTAINER_ID>
#
docker pause <CONTAINER_ID>
#
docker unpause <CONTAINER_ID>
#
docker restart <CONTAINER_ID>
# Авторизоваться в акк dockerhub
docker login
